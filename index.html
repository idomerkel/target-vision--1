// ========================================== // PARALLEL KINEMATIC DIAGNOSTIC ENGINE (IMU) // ========================================== const KINEMATIC_BATCH_SIZE = 25; let kinematicBuffer = []; let latestPhysicalMistake = null; let lastShotTime = 0; const GYRO_SHOT_THRESHOLD = 80; const ACCEL_SHOT_THRESHOLD = 15; const GYRO_MOVE_THRESH = 10.0; const ACCEL_MOVE_THRESH = 1.5; const mistakeDB = [ { name: "Jerking the trigger", patterns: ["Down", "Pitch"] }, { name: "Flinching/anticipating", patterns: ["Down", "Pitch"] }, { name: "Inconsistent finger placement", patterns: ["Left", "Yaw", "Right"] }, { name: "Poor grip/stance", patterns: ["Left", "Right", "Roll"] }, { name: "Not following through", patterns: ["Down", "Pitch"] }, { name: "Canting firearm", patterns: ["Roll"] }, { name: "Fatigue/shaking", patterns: ["Up", "Down", "Left", "Right"] } ]; function enableSensors() { if (typeof DeviceMotionEvent.requestPermission === 'function') { DeviceMotionEvent.requestPermission() .then(permissionState => { if (permissionState === 'granted') { window.addEventListener('devicemotion', handleMotion); document.getElementById('status').innerText = "Session & Sensors Active."; } else { document.getElementById('status').innerText = "Sensor Access Denied by Phone."; } }) .catch(err => { console.error(err); document.getElementById('status').innerText = "Sensor Error (Needs HTTPS)."; }); } else { // Standard Android behavior window.addEventListener('devicemotion', handleMotion); document.getElementById('status').innerText = "Session & Sensors Active."; } } function handleMotion(event) { if (!isTracking) return; // CRITICAL FIX: The '?.' prevents crashes if the phone's sensor isn't ready let ax = event.acceleration?.x || 0; let ay = event.acceleration?.y || 0; let az = event.acceleration?.z || 0; let gx = event.rotationRate?.alpha || 0; let gy = event.rotationRate?.beta || 0; let gz = event.rotationRate?.gamma || 0; let now = Date.now(); kinematicBuffer.push({ax, ay, az, gx, gy, gz}); if (kinematicBuffer.length > KINEMATIC_BATCH_SIZE) { kinematicBuffer.shift(); } if (now - lastShotTime > 500) { if (Math.abs(gx) > GYRO_SHOT_THRESHOLD || Math.abs(gy) > GYRO_SHOT_THRESHOLD || Math.abs(ax) > ACCEL_SHOT_THRESHOLD) { lastShotTime = now; analyzePreShotWindow(); } } } function analyzePreShotWindow() { let down = false, up = false, left = false, right = false; let pitch = false, yaw = false, roll = false; let moveString = ""; let preShotFrames = kinematicBuffer.slice(0, KINEMATIC_BATCH_SIZE - 5); for (let s of preShotFrames) { if (s.ax > ACCEL_MOVE_THRESH) right = true; if (s.ax < -ACCEL_MOVE_THRESH) left = true; if (s.az > ACCEL_MOVE_THRESH) up = true; if (s.az < -ACCEL_MOVE_THRESH) down = true; if (Math.abs(s.gx) > GYRO_MOVE_THRESH) roll = true; if (Math.abs(s.gy) > GYRO_MOVE_THRESH) pitch = true; if (Math.abs(s.gz) > GYRO_MOVE_THRESH) yaw = true; } if (down) moveString += "Down,"; if (up) moveString += "Up,"; if (left) moveString += "Left,"; if (right) moveString += "Right,"; if (pitch) moveString += "Pitch,"; if (yaw) moveString += "Yaw,"; if (roll) moveString += "Roll,"; let bestMatch = null; let maxScore = 0; for (let mistake of mistakeDB) { let score = 0; for (let pattern of mistake.patterns) { if (moveString.includes(pattern)) score += 10; } if (score > maxScore) { maxScore = score; bestMatch = mistake.name; } } if (bestMatch && maxScore > 0) { latestPhysicalMistake = bestMatch; } }
