function startCamera() { // Request the environment (rear) camera for target shooting navigator.mediaDevices.getUserMedia({ video: { facingMode: "environment", width: 640, height: 480 } }) .then(s => { const v = document.getElementById('webcam'); v.srcObject = s; v.onloadedmetadata = () => { const ov = document.getElementById('overlayCanvas'); // CRITICAL ALIGNMENT FIX: // Force the overlay to match the 640x480 OpenCV math canvas, // NOT the raw hardware camera resolution! ov.width = 640; ov.height = 480; document.getElementById('btnBase').disabled = false; document.getElementById('btnCam').style.display = 'none'; document.getElementById('status').innerText = "Camera Active."; }; }).catch(e => { document.getElementById('status').innerText = "Camera Error."; console.error(e); }); }
